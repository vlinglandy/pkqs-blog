class Solution {
public:
    int n;
    string ss;
    void fn(vector<int>& v,int i){
        if(i<0||i>=n)return ;
        int index=i;
        i--;
        while(i>=0&&i<n){
        	//cout<<i<<endl;
            if(v[i]!=0){
            	//cout<<index<<endl;
            	if(i>0&&i<n-1){
                	v[i]=(v[i+1]<v[i-1]?v[i+1]:v[i-1])+1;
	            }else if(i==0){
	                v[i]=v[i+1]+1;
	            }else if(i==n-1){
	                v[i]=v[i-1]+1;
	            }else{
                    break;
                }
	            i=i--;
	            if(i==-1)return ;
	            cout<<i<<endl;
			}else break;
        }
        index++;
        while(index>=0&&index<n){
        	//cout<<index<<endl;
            if(v[index]!=0){
            	//cout<<index<<endl;
            	if(index>0&&index<n-1){
                	v[i]=(v[index+1]<v[index-1]?v[index+1]:v[index-1])+1;
	            }else if(index==0){
	                v[index]=v[index+1]+1;
	            }else if(index==n-1){
	                v[index]=v[index-1]+1;
	            }else{
                    break;
                }
	            index++;
			}else break;
        }

    }
    vector<int> shortestToChar(string s, char c) {
        ss=s;
        queue<int> que;
        n=s.length();
        cout<<n<<endl;
        vector<int> v(20,20);
//        for(int i=0;i<v.size();i++){

        for(int i=0;i<n;i++){
            if(s[i]==c){
                v[i]=0;
                que.push(i);
            }
        }
        while(!que.empty()){
            fn(v,que.front());
            que.pop();
        }
        return v;
    }
};
